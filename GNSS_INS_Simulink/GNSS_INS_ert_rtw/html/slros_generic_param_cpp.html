5/**



6 * Initialize the parameter getter class.



7 * @param pName The name of the ROS parameter



8 */



9void SimulinkParameterGetterBase::initialize(const std::string& pName) {



10    nodePtr = SLROSNodePtr;



11    paramName = pName;



12    hasValidValue = false;



13}



14



15/**



16 * Initialize the constants for the error codes.



17 * @param codeSuccess Error code that should be emitted if parameter is retrieved successfully



18 * @param codeNoParam Error code if parameter with given name does not exist on server



19 * @param codeTypeMismatch Error code if parameter exists on server, but has a different data type



20 * @param codeArrayTruncate Error code if received array was truncated



21 */



22void SimulinkParameterGetterBase::initialize_error_codes(uint8_t codeSuccess,



23                                                         uint8_t codeNoParam,



24                                                         uint8_t codeTypeMismatch,



25                                                         uint8_t codeArrayTruncate) {



26    // Initialize the error codes



27    errorCodeSuccess = codeSuccess;



28    errorCodeNoParam = codeNoParam;



29    errorCodeTypeMismatch = codeTypeMismatch;



30    errorCodeArrayTruncate = codeArrayTruncate;



31}
