78int mw_CreateArmedTimer(double periodInSeconds)



79{



80    int status;



81    int fd;



82    struct itimerspec its;



83



84    /* Create the timer */



85    fd = timerfd_create(CLOCK_MONOTONIC, 0);



86    if (fd == -1) {



87        fprintf(stderr, "Call to timerfd_create failed.\n"); 



88        perror("timerfd_create");



89        fflush(stderr); 



90        exit(EXIT_FAILURE);



91    }



92



93    /* Make the timer periodic */



94    its.it_value.tv_sec = (time_t)periodInSeconds;



95    its.it_value.tv_nsec = (periodInSeconds - (time_t)periodInSeconds) * 1000000000;



96    its.it_interval.tv_sec = its.it_value.tv_sec;



97    its.it_interval.tv_nsec = its.it_value.tv_nsec;



98    status = timerfd_settime(fd, 0, &its, NULL);



99    CHECK_STATUS(status, 0, "timer_settime");



100    



101    return fd;



102}

164void mw_WaitForTimerEvent(int fd)



165{



166    unsigned long long missed;



167    int status;



168    



169    /* Wait for the next timer event. If we have missed any the



170       number is written to "missed" */



171    while ((status = read(fd, &missed, sizeof(missed)) == -1) && (errno == EINTR)) {



172        /* Restart if interrupted by a signal */



173        continue;



174    }



175    if (status == -1) {



176        perror("read(timerfd)");



177    }



178}



179



180void mw_WaitForTimerEventCatchup(int fd)



181{



182    unsigned long long missed = 0;



183    int status = 0;



184    static unsigned int outstanding = 0;



185    



186    /* Wait for the next timer event. If we have missed any the



187     * number is written to "missed" */



188    if (outstanding == 0)



189    {



190        while ((status = read(fd, &missed, sizeof(missed)) == -1) && (errno == EINTR)) {



191            /* Restart if interrupted by a signal */



192            continue;



193        }



194        if (status == -1) {



195            perror("read(timerfd)");



196        }



197        if (missed > 1) {



198            #ifdef MW_RTOS_DEBUG



199            printf("Missed %llu events for the timer # %d.\n", missed, fd);



200            fflush(stdout);



201            #endif



202            outstanding = outstanding + missed - 1;



203        }



204    }



205    else



206    {



207        #ifdef MW_RTOS_DEBUG



208        printf("Catching up with the missed events for timer # %d.\n", fd);



209        fflush(stdout);



210        #endif



211        outstanding--;



212    }



213}

283void mw_CreateTask(void * (*taskHandler)(void *), const char* taskName, int priority, int policy, int coreSelection, int coreNum)



284{



285    int status;



286    int inherit;



287    pthread_attr_t attr;



288    pthread_t thread;



289    struct sched_param param;



290    size_t stackSize;



291    pthread_attr_init(&attr);



292    cpu_set_t cpuset;



293#ifdef MW_TSKMGR_TIMER_DRIVEN_TASKS



294    char thisTaskName[MW_MAX_TASKNAME];



295#endif



296



297    /* Set thread inherit attribute */



298    inherit = PTHREAD_EXPLICIT_SCHED;



299    status = pthread_attr_setinheritsched(&attr, inherit);



300    CHECK_STATUS(status, 0, "pthread_attr_setinheritsched");



301



302    /* Set thread detach attribute */



303    status = pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_DETACHED);



304    CHECK_STATUS(status, 0, "pthread_attr_setdetachstate");



305



306    /* Set thread stack size attribute */



307    stackSize = (512 > PTHREAD_STACK_MIN) ? 512:PTHREAD_STACK_MIN;



308    status = pthread_attr_setstacksize(&attr, stackSize);



309    CHECK_STATUS(status, 0, "pthread_attr_setstacksize");



310



311    /* Set thread schedule policy attribute */



312    policy = SCHED_FIFO;



313    status = pthread_attr_setschedpolicy(&attr, policy);



314    CHECK_STATUS(status, 0, "pthread_attr_setschedpolicy");



315



316    /* Set thread priority attribute */



317    param.sched_priority = priority;



318    status = pthread_attr_setschedparam(&attr, &param);



319    CHECK_STATUS(status, 0, "pthread_attr_setschedparam");



320    



321    /* Set the thread core affinity */



322    if (2 == coreSelection) 



323    {



324        CPU_ZERO(&cpuset);



325        CPU_SET(coreNum, &cpuset);



326        status =  pthread_attr_setaffinity_np(&attr, sizeof(cpu_set_t), &cpuset);



327        CHECK_STATUS(status, 0, "pthread_attr_setaffinity_np");



328    }



329



330    /* Create the thread */



331    status = pthread_create(&thread, &attr, taskHandler, NULL);



332    CHECK_STATUS(status, 0, "pthread_create");



333#ifdef MW_TSKMGR_TIMER_DRIVEN_TASKS



334    /* Set name of the thread */



335    snprintf(thisTaskName, MW_MAX_TASKNAME, "%s", taskName);



336    status = pthread_setname_np(thread, thisTaskName);



337    CHECK_STATUS(status, 0, "pthread_setname_np at mw_CreateTask");



338#endif



339    pthread_attr_destroy(&attr);



340}

342void myAddBlockForThisEvent(int sigNo)



343{



344    int status;



345    sigset_t sigMask;



346



347    sigemptyset(&sigMask);



348    sigaddset(&sigMask, sigNo);



349    status = pthread_sigmask(SIG_BLOCK, &sigMask, NULL);



350    CHECK_STATUS(status, 0, "pthread_sigmask");



351}



352



353void myAddHandlerForThisEvent(int sigNo, int sigToBlock[], int numSigToBlock, void (*sigHandler)(int))



354{



355    int idx;



356    int status;



357    struct sigaction sa;



358



359    sa.sa_handler = (__sighandler_t) sigHandler;



360    sigemptyset(&sa.sa_mask);



361    for (idx=0; idx<numSigToBlock; idx++) {



362            sigaddset(&sa.sa_mask, sigToBlock[idx]);



363    }



364    sa.sa_flags = SA_RESTART; /* Restart functions if interrupted by handler */



365    status = sigaction(sigNo, &sa, NULL);



366    CHECK_STATUS_NOT(status, -1, "sigaction to register a signal handler");



367}



368



369void myRestoreDefaultHandlerForThisEvent(int sigNo)



370{



371    int status;



372    struct sigaction sa;



373    sa.sa_handler = SIG_DFL;



374    sigemptyset(&sa.sa_mask);



375    sa.sa_flags = SA_RESTART; /* Restart functions if interrupted by handler */



376    status = sigaction(sigNo, &sa, NULL);



377    CHECK_STATUS_NOT(status, -1, "sigaction to restore default signal handler");



378}

384void myRTOSInit(double baseRatePeriod, int numSubrates)



385{



386    int i;



387    int status;



388    uid_t euid;



389    size_t stackSize;



390    unsigned long cpuMask = 0x1;



391    unsigned int len = sizeof(cpuMask);  



392#ifdef MW_TSKMGR_TIMER_DRIVEN_TASKS



393    char thisTaskName[MW_MAX_TASKNAME];



394#endif



395



396#if (defined(MW_TSKMGR_EVENT_DRIVEN_TASKS) && (MW_TSKMGR_EVENT_DRIVEN_TASKS > 0))



397    {



398        int idx;



399        for (idx=0;idx < MW_TSKMGR_EVENT_DRIVEN_TASKS;idx++)



400        {



401            mw_CreateTask(asyncTaskCodeGenInfo[idx].pAsyncTasks, \



402                    (const char*) asyncTaskCodeGenInfo[idx].taskName,\



403                    asyncTaskCodeGenInfo[idx].priority, \



404                    asyncTaskCodeGenInfo[idx].policy, \



405                    asyncTaskCodeGenInfo[idx].coreSelection, \



406                    asyncTaskCodeGenInfo[idx].coreNum);



407        }



408    }



409#endif



410    UNUSED(baseRatePeriod);



411    UNUSED(numSubrates);



412        



413    if (!MW_IS_CONCURRENT) {



414        /* All threads created by this process will run on a single CPU */



415        status = sched_setaffinity(0, len, (cpu_set_t *) &cpuMask);



416        CHECK_STATUS(status, 0, "sched_setaffinity");



417    }



418



419#if MW_SP_SCHED_FIFO && !defined (_POSIX_THREAD_PRIORITY_SCHEDULING)



420    fprintf(stderr, "Priority scheduling is NOT supported by your system.\n");



421    fprintf(stderr, "The generated code will not run correctly because your\n");



422    fprintf(stderr, "model contains multiple rates and uses multi-tasking\n");



423    fprintf(stderr, "code generation mode. You can only run the generated code\n");



424    fprintf(stderr, "in single-tasking mode in your system. Open\n");



425    fprintf(stderr, "Simulation -> Configuration Parameters -> Solver dialog\n");



426    fprintf(stderr, "and set \"Tasking mode for periodic sample times\" parameter to SingleTasking.\n");



427    fprintf(stderr, "Re-build the Simulink model with the new settings and try executing the generated code again.\n");



428    fflush(stderr);



429    exit(EXIT_FAILURE);



430#endif



431    



432#if MW_SP_SCHED_FIFO



433    /* Need root privileges for real-time scheduling */



434    euid = geteuid();



435    if (euid != 0) {



436        fprintf(stderr, "You must have root privileges to run the generated code because\n");



437        fprintf(stderr, "generated code requires SCHED_FIFO scheduling class to run correctly.\n");



438        fprintf(stderr, "Try running the executable with the following command: sudo ./<executable name>\n");



439        fflush(stderr);



440        exit(EXIT_FAILURE);



441    }



442#endif



443



444    status = sem_init(&baserateTaskSem, 0, 0);



445    CHECK_STATUS(status, 0, "sem_init:baserateTaskSemSem");



446    status = sem_init(&stopSem, 0, 0);



447    CHECK_STATUS(status, 0, "sem_init:stopSem");



448    



449#if MW_SP_SCHED_FIFO



450    /* Set scheduling policy of the main thread to SCHED_FIFO */



451    sp.sched_priority = sched_get_priority_max(SCHED_FIFO);



452    status = sched_setscheduler(0, SCHED_FIFO, &sp);



453    CHECK_STATUS(status, 0, "sched_setscheduler");



454#endif



455



456    /* Create threads executing the Simulink model */



457    pthread_attr_init(&attr);



458    status = pthread_attr_setinheritsched(&attr, PTHREAD_EXPLICIT_SCHED);



459    CHECK_STATUS(status, 0, "pthread_attr_setinheritsched");



460#if MW_SP_SCHED_FIFO



461    status = pthread_attr_setschedpolicy(&attr, SCHED_FIFO);



462#else



463    status = pthread_attr_setschedpolicy(&attr, SCHED_OTHER);



464#endif



465    CHECK_STATUS(status, 0, "pthread_attr_setschedpolicy");



466    status = pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_JOINABLE);



467    CHECK_STATUS(status, 0, "pthread_attr_setdetachstate");



468



469    /* Set thread stack size if necessary */



470    status = pthread_attr_getstacksize(&attr, &stackSize);



471    CHECK_STATUS(status, 0, "pthread_attr_getstacksize");



472    if (stackSize < STACK_SIZE) {



473        /* Make sure that stackSize is a multiple of 8 */



474        stackSize = (STACK_SIZE + 7) & (~0x7);



475        pthread_attr_setstacksize(&attr, stackSize);



476        CHECK_STATUS(status, 0, "pthread_attr_setstacksize");



477    }



478



479    signal(SIGTERM, exitFcn);     /* kill */



480    signal(SIGHUP, exitFcn);      /* kill -HUP */



481    signal(SIGINT, exitFcn);      /* Interrupt from keyboard */



482    signal(SIGQUIT, exitFcn);     /* Quit from keyboard */



483



484#ifdef MW_STANDALONE_EXECUTION_PROFILER_ON



485    status = pthread_mutex_init(&profilingDataStoreMutex, NULL);    



486#endif



487    



488#ifdef MW_HAS_MULTIPLE_RATES



489    MW_DEBUG_LOG("**creating subrate task threads**\n");   



490    for (i = 0; i < MW_NUMBER_SUBRATES; i++) {



491        taskId[i] = i;



492        status = sem_init(&subrateTaskSem[i], 0, 0);



493        CHECK_STATUS(status, 0, "sem_init");



494        setThreadPriority(subratePriority[i], &attr, &sp);



495        status = pthread_create(&subRateThread[i], &attr, &subrateTask, (void *)&taskId[i]);



496        CHECK_STATUS(status, 0, "pthread_create");



497#ifdef MW_TSKMGR_TIMER_DRIVEN_TASKS



498        /* Set name of sub-rate threads */



499        snprintf(thisTaskName, MW_MAX_TASKNAME, "%s", _mwTskMgrTimerDrivenTaskNames[i+1]);



500        MW_DEBUG_LOG(thisTaskName);   



501        status = pthread_setname_np(subRateThread[i], thisTaskName);



502        CHECK_STATUS(status, 0, "pthread_setname_np at subRateThread");



503#endif        



504#ifdef DETECT_OVERRUNS



505        status = pthread_mutex_init(&rateTaskFcnRunningMutex[i+1], NULL);



506        CHECK_STATUS(status, 0, "pthread_mutex_init");



507#endif        



508#ifdef COREAFFINITYREQUIRED



509        if (coreAffinity[i] >= 0) {



510             cpu_set_t cpuset;



511             CPU_ZERO(&cpuset);



512             CPU_SET(coreAffinity[i], &cpuset);



513             status = pthread_setaffinity_np(subRateThread[i], sizeof(cpu_set_t), &cpuset);



514             CHECK_STATUS(status, 0, "pthread_setaffinity_np");



515         }



516#endif



517    }



518#endif



519



520    MW_DEBUG_LOG("**creating the base rate task thread**\n");    



521    setThreadPriority(MW_BASERATE_PRIORITY, &attr, &sp);  



522    status = pthread_create(&baseRateThread, &attr, &baseRateTask, NULL);



523    CHECK_STATUS(status, 0, "pthread_create");



524    



525#ifdef MW_TSKMGR_TIMER_DRIVEN_TASKS



526    /* Set name of the base-rate thread */



527    snprintf(thisTaskName, MW_MAX_TASKNAME, _mwTskMgrTimerDrivenTaskNames[0]);



528    status = pthread_setname_np(baseRateThread, thisTaskName);



529    CHECK_STATUS(status, 0, "pthread_setname_np at baserateThread");



530#endif



531    



532#ifdef DETECT_OVERRUNS



533    status = pthread_mutex_init(&rateTaskFcnRunningMutex[0], NULL);



534    CHECK_STATUS(status, 0, "pthread_mutex_init");



535#endif



536



537#ifdef COREAFFINITYREQUIRED



538    if (coreAffinityBaseRate >= 0) {



539        cpu_set_t cpuset;



540        CPU_ZERO(&cpuset);



541        CPU_SET(coreAffinityBaseRate, &cpuset);



542        status = pthread_setaffinity_np(baseRateThread, sizeof(cpu_set_t), &cpuset);



543        CHECK_STATUS(status, 0, "pthread_setaffinity_np");



544    }   



545#endif



546



547#ifdef MW_SOC_ENABLED    



548    MW_DEBUG_LOG("**creating the socb base rate timer handler thread**\n");  



549    /* Set the priority higher (higher number) than the base rate */    



550    setThreadPriority(MW_BASERATE_PRIORITY + 2, &attr, &sp);



551    info.period = MW_BASERATE_PERIOD;



552    status = pthread_create(&baseTimerThread, &attr, &baseTimerTask, (void *) &info);



553    CHECK_STATUS(status, 0, "pthread_create");



554    status = pthread_setname_np(baseTimerThread, "baseTimerTask");



555#endif 



556    



557    MW_DEBUG_LOG("**creating the scheduler thread**\n");  



558    /* Set the priority higher (higher number) than the base rate */    



559    setThreadPriority(MW_BASERATE_PRIORITY + 1, &attr, &sp);



560    info.period = MW_BASERATE_PERIOD;



561    status = pthread_create(&schedulerThread, &attr, &schedulerTask, (void *) &info);



562    CHECK_STATUS(status, 0, "pthread_create");



563       



564#ifdef MW_TSKMGR_TIMER_DRIVEN_TASKS



565    /* Set name of the scheduler thread */



566    snprintf(thisTaskName, MW_MAX_TASKNAME, "scheduler");



567    status = pthread_setname_np(schedulerThread, thisTaskName);



568    CHECK_STATUS(status, 0, "pthread_setname_np at scheduler thread");



569#endif



570



571#ifdef MW_HAS_APERIODIC_TASKS



572    MW_DEBUG_LOG("**creating asynchronously triggered task threads**\n"); 



573    /* Set the priority higher (higher number) than the base rate */    



574    sp.sched_priority = MW_BASERATE_PRIORITY + 1;



575    for (i = 0; i < MW_NUMBER_APERIODIC_TASKS; i++) {



576        status = pthread_create(&asyncThread[i], &attr, (void *) pAsyncTasks[i], NULL);



577        CHECK_STATUS(status, 0, "pthread_create");



578    }



579#endif



580    



581#ifdef MW_NEEDS_BACKGROUND_TASK



582    MW_DEBUG_LOG("**creating the background thread**\n");



583    status = pthread_attr_setschedpolicy(&attr, SCHED_OTHER);



584    CHECK_STATUS(status, 0, "pthread_attr_setschedpolicy");



585    setThreadPriority(0, &attr, &sp);



586    status = pthread_create(&backgroundThread, &attr, &backgroundTask, NULL);



587    CHECK_STATUS(status, 0, "pthread_create");



588#ifdef MW_TSKMGR_TIMER_DRIVEN_TASKS

591    status = pthread_setname_np(backgroundThread, thisTaskName);



592    CHECK_STATUS(status, 0, "pthread_setname_np at backgroundThread");



593#endif 



594#if MW_SP_SCHED_FIFO == 0



595    status = pthread_setschedparam(backgroundThread, SCHED_IDLE, &sp);



596    CHECK_STATUS(status, 0, "pthread_setschedparam");



597#ifdef MW_HAS_COMM_SERVICE



598    status = makeCSTaskIdle();



599    CHECK_STATUS(status, 0, "pthread_setschedparam");



600#endif 



601#endif



602#endif



603



604    pthread_attr_destroy(&attr);



605    fflush(stdout);



606}
